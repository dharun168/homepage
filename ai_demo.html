This demo provides a compact C program that trains a single-layer perceptron to learn the AND logic gate. Download the source, compile with gcc, and run locally.

// Activation: step function
int activate(double x) {
  return x >= 0.0 ? 1 : 0;
}

int main(void) {
  // Training data for AND gate
  double X[4][2] = {{0,0},{0,1},{1,0},{1,1}};
  int Y[4] = {0,0,0,1};

  double w0 = 0.0, w1 = 0.0, b = 0.0; // weights and bias
  double lr = 0.1; // learning rate
  int epochs = 50;

  srand((unsigned)time(NULL));
  // small random init
  w0 = (rand()/(double)RAND_MAX - 0.5);
  w1 = (rand()/(double)RAND_MAX - 0.5);
  b  = (rand()/(double)RAND_MAX - 0.5);

  for (int e = 0; e < epochs; ++e) {
    int total_err = 0;
    for (int i = 0; i < 4; ++i) {
      double x0 = X[i][0];
      double x1 = X[i][1];
      double z = w0*x0 + w1*x1 + b;
      int yhat = activate(z);
      int err = Y[i] - yhat;
      if (err != 0) {
        w0 += lr * err * x0;
        w1 += lr * err * x1;
        b  += lr * err;
      }
      total_err += abs(err);
    }
    // print progress every 10 epochs
    if (e % 10 == 0) printf("epoch %d  error=%d\n", e, total_err);
    if (total_err == 0) break;
  }

  printf("\nTrained weights:\n  w0=%.4f\n  w1=%.4f\n  b=%.4f\n\n", w0, w1, b);
  printf("Predictions:\n");
  for (int i = 0; i < 4; ++i) {
    double x0 = X[i][0];
    double x1 = X[i][1];
    int yhat = activate(w0*x0 + w1*x1 + b);
    printf("[%g, %g] -> %d\n", x0, x1, yhat);
  }
  return 0;
}

What it does

This program trains a single perceptron with a step activation to classify the AND truth table. It's a compact example of supervised learning implemented in plain C.
